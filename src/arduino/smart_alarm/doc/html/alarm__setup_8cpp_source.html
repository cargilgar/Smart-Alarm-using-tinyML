33 void getwakeUpTimeRange(const char* currentTime, const char* wakeUpTime, uint8_t interval, unsigned long* arrRange) {
34 
35  unsigned long time1 = charToSeconds(currentTime);
36  unsigned long time2 = charToSeconds(wakeUpTime);
37 
38  // Add 24 hours (86400s) when time2 is the next day.
39  // For example: (time1 = 23:00, time2 = 07:00)
40  if(time1 > time2)
41  time2 += 86400;
42 
43  arrRange[0] = (time2 - time1 - interval * 60 / 2) * 1000;
44  arrRange[1] = (time2 - time1 + interval * 60 / 2) * 1000;
45 }

47 unsigned long charToSeconds(const char* time) {
48

53  acc += (time[3] - '0') * 600;
