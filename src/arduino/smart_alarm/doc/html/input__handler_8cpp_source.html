24 void InputHandler::generateFeatures(float x, float y, float z, float bpm) {
25 
26  if(!_initialized) {
27  features[0] = x; // x axis
28  features[1] = y; // y axis
29  features[2] = z; // z axis
30  _initialized = true;
31  }
32  else {
33  features[3] = bpm;
34 
35  features[4] = x - features[0]; // max_value - last max_value, x axis
36  features[5] = y - features[1]; // max_value - last max_value, y axis
37  features[6] = z - features[2]; // max_value - last max_value, z axis
38 
39 
40  features[0] = x; // x axis
41  features[1] = y; // y axis
42  features[2] = z; // z axis
43 
44  features[7] = sqrt(sq(x) + sq(y) + sq(z)); // max_value module
45 
46  features[8] = sqrt(sq(features[4]) + sq(features[5]) + sq(features[6])); // module 'acc subtraction'
47 
48  features[9] = x * x * x; // max_x cubed (X**3)
49 
50  features[10] = sq(features[4]); // (max_value - last max_value) squared, x axis
51 
52  features[11] = sq(features[6]); // (max_value - last max_value) squared, z axis
53 
54  features[12] = sq(features[8]); // module 'acc subtraction' squared
55 
56  features[13] = pow(e, features[4]); // exp(max_x - last max_x)
57 
58  _initialized = false;
59 
60  _normalizeFeatures();
61  }
62 }
