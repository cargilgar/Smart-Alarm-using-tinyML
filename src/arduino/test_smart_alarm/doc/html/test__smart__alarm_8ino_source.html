55void setup() {
56
57 Serial.begin(9600);
58
59 // Setting up logging
60 error_reporter = &micro_error_reporter;
61
62 // Map the model into a usable data structure.
63 model = tflite::GetModel(g_model);
64
65 // Check the model has been converted with a compatible tflite converter version
66 if (model->version() != TFLITE_SCHEMA_VERSION){
67 TF_LITE_REPORT_ERROR(error_reporter,
68 "Model provided is schema version %d not equal "
69 "to supported version %d.",
70 model->version(), TFLITE_SCHEMA_VERSION);
71 return;
72 }
73
74 static tflite::AllOpsResolver resolver;
75 // static tflite::MicroMutableOpResolver resolver(error_reporter);
76
77 // Instantiate the interpreter to run the model with.
78 static tflite::MicroInterpreter static_interpreter(
79 model, resolver, tensor_arena, kTensorArenaSize, error_reporter);
80
81 interpreter = &static_interpreter;
82
83 // Allocate memory from the tensor_arena for the model's tensors.
84 TfLiteStatus allocate_status = interpreter->AllocateTensors();
85
86 if (allocate_status != kTfLiteOk){
87 TF_LITE_REPORT_ERROR(error_reporter, "AllocateTensors() failed");
88 return;
89 }
90
91 // Set model's input and output.
92 model_input = interpreter->input(0);
93 model_output = interpreter->output(0);
94
95 Serial.println("Please enter any key to start the test.");
96}

99void loop()
100{
101 while(Serial.available())
102 {
103 for(;;){
104 int incommingByte = Serial.read();
105
106 if(incommingByte != 0)
107 break;
108 }
109 }
110
111 for(uint8_t i = 0; i < kNumTests; i++){
112
113 testInputTensor();
114
115 // Popullate model input from samples
116 for (int8_t j = 0; j < kFeatureCount; j++) {
117 // int8_t quantized = samples[j+i*14] / model_input->params.scale + model_input->params.zero_point;
118 int8_t quantized = quantize(samples[j+i*14]);
119 model_input->data.int8[j] = quantized;
120 }
121
122 // Take the 15th value for the label
123 label_test = samples[i*15];
124
125 // Run inference, and report any error
126 TfLiteStatus invoke_status = interpreter->Invoke();
127
128 if (invoke_status != kTfLiteOk)
129 TF_LITE_REPORT_ERROR(error_reporter, "Invoke failed\n");
130
131 testOutputTensor();
132
133 for(int i = 0; i < kLabelCount; i++)
134 output_array[i] = model_output->data.int8[i];
135
136 int8_t prediction = recognizeLabel(output_array, true);
137
138 if(prediction == label_test)
139 correctPredicition++;
140 }
141
142 TF_LITE_REPORT_ERROR(error_reporter, "Test finalized successfully!"
143 "\nNumber of correct predicitions: %d over %d samples.\n\n",
144 correctPredicition, kNumTests);
145
146 correctPredicition = 0;
147}

152void testInputTensor(){
153 if ((model_input == nullptr))
154 TF_LITE_REPORT_ERROR(error_reporter, "Input Tensor does not exist.\n");
155
156 if ((model_input->dims->size != 2))
157 TF_LITE_REPORT_ERROR(error_reporter, "Input Tensor size incorrect.\n");
158
159 if ((model_input->dims->data[0] != 1))
160 TF_LITE_REPORT_ERROR(error_reporter, "The first dimension of the Input "
161 "Tensor is not %d.It is: %d.\n", 1, model_input->dims->data[0]);
162
163 if ((model_input->dims->data[1] != kFeatureCount))
164 TF_LITE_REPORT_ERROR(error_reporter, "The second dimension of the Input "
165 "Tensor is not %d.It is: %d.\n", kFeatureCount, model_input->dims->data[1]);
166
167 if ((model_input->type != kTfLiteInt8))
168 TF_LITE_REPORT_ERROR(error_reporter, "Input Tensor type is not %d.\n",
169 kTfLiteInt8);
170}
