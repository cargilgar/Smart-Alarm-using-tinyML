20int InputHandler::generateFeatures(float x, float y, float z, float bpm){
21
22 if(!_initialized){
23 features[0] = x; // x axis
24 features[1] = y; // y axis
25 features[2] = z; // z axis
26 _initialized = true;
27 return 1;
28 }
29
30 features[3] = bpm;
31
32 features[4] = x - features[0]; // max_value - last max_value, x axis
33 features[5] = y - features[1]; // max_value - last max_value, y axis
34 features[6] = z - features[2]; // max_value - last max_value, z axis
35
36
37 features[0] = x; // x axis
38 features[1] = y; // y axis
39 features[2] = z; // z axis
40
41 features[7] = sqrt(sq(x) + sq(y) + sq(z)); // max_value module
42
43 features[8] = sqrt(sq(features[4]) + sq(features[5]) + sq(features[6])); // module 'acc subtraction'
44
45 features[9] = x * x * x; // max_x cubed (X**3)
46
47 features[10] = sq(features[4]); // (max_value - last max_value) squared, x axis
48
49 features[11] = sq(features[6]); // (max_value - last max_value) squared, z axis
50
51 features[12] = sq(features[8]); // module 'acc subtraction' squared
52
53 features[13] = pow(e, features[4]); // exp(max_x - last max_x)
54
55 _initialized = false;
56
57 _normalizeFeatures();
58
59 return 0;
60}
