68void setup() {
69 // Start serial
70 Serial.begin(9600);
71 Serial.println("Started");
72
73 // Setting up logging
74 error_reporter = &micro_error_reporter;
75
76 // Map the model into a usable data structure.
77 model = tflite::GetModel(g_model);
78
79 // Check the model has been converted with a compatible tflite converter version
80 if (model->version() != TFLITE_SCHEMA_VERSION)
81 {
82 TF_LITE_REPORT_ERROR(error_reporter,
83 "Model provided is schema version %d not equal "
84 "to supported version %d.",
85 model->version(), TFLITE_SCHEMA_VERSION);
86 return;
87 }
88
89 static tflite::AllOpsResolver resolver;
90
91 // Instantiate the interpreter to run the model with.
92 static tflite::MicroInterpreter static_interpreter(
93 model, resolver, tensor_arena, kTensorArenaSize, error_reporter);
94
95 interpreter = &static_interpreter;
96
97 // Allocate memory from the tensor_arena for the model's tensors.
98 TfLiteStatus allocate_status = interpreter->AllocateTensors();
99
100 if (allocate_status != kTfLiteOk){
101 TF_LITE_REPORT_ERROR(error_reporter, "AllocateTensors() failed");
102 return;
103 }
104
105 // Set model's input and output settings
106 model_input = interpreter->input(0);
107 model_output = interpreter->output(0);
108
109}

182void testInputTensor(){
183 // TEST INPUT TENSOR
184 if ((model_input == nullptr))
185 TF_LITE_REPORT_ERROR(error_reporter, "Input Tensor does not exist.\n");
186
187 if ((model_input->dims->size != 2))
188 TF_LITE_REPORT_ERROR(error_reporter, "Input Tensor size incorrect.\n");
189
190 if ((model_input->dims->data[0] != 1))
191 TF_LITE_REPORT_ERROR(error_reporter, "The first dimension of the Input "
192 "Tensor is not %d.It is: %d.\n", 1, model_input->dims->data[0]);
193
194 if ((model_input->dims->data[1] != kFeatureCount))
195 TF_LITE_REPORT_ERROR(error_reporter, "The second dimension of the Input "
196 "Tensor is not %d.It is: %d.\n", kFeatureCount, model_input->dims->data[1]);
197
198 if ((model_input->type != kTfLiteInt8))
199 TF_LITE_REPORT_ERROR(error_reporter, "Input Tensor type is not %d.\n",
200 kTfLiteInt8);
201}
